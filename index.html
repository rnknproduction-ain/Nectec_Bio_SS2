Execution Notes จาก TOR

  • ต้องคำนึงถึงความสะดวก ความปลอดภัย และสิทธิของผู้เข้าร่วม
  • สถานที่ควรเหมาะกับข้อมูลทั้ง 7 ประเภท และมี privacy เพียงพอ
  • ข้อมูลผิดพลาดหรือไม่สมบูรณ์ต้องเก็บใหม่ให้ครบถ้วน
  • ต้องมีโครงสร้างไฟล์, การตั้งชื่อ, zip file และ password/encryption เพื่อป้องกันการเข้าถึงจากผู้ไม่ได้รับอนุญาต
  • งวดที่ 1 ต้องส่งแผนดำเนินการ Gantt Chart, บุคลากร, สถานที่, ขั้นตอนสรรหา, โครงสร้างไฟล์ และข้อมูลจริง 100 คน

Online Save Setup

สร้าง Google Sheet → Extensions → Apps Script → วางโค้ด → Deploy as Web App → Copy URL มาใส่ช่อง Web App URL

const SHEET_NAME='NECTEC_Feasibility_Log';
function doPost(e){const body=JSON.parse(e.postData.contents||'{}');const ss=SpreadsheetApp.getActiveSpreadsheet();let sh=ss.getSheetByName(SHEET_NAME);if(!sh){sh=ss.insertSheet(SHEET_NAME);sh.appendRow(['timestamp','updatedBy','payload']);}if(body.action==='save'){sh.appendRow([new Date(),body.updatedBy||'',JSON.stringify(body.payload||{})]);return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);}return ContentService.createTextOutput(JSON.stringify({ok:false})).setMimeType(ContentService.MimeType.JSON);}
function doGet(){const ss=SpreadsheetApp.getActiveSpreadsheet();const sh=ss.getSheetByName(SHEET_NAME);if(!sh||sh.getLastRow()<2)return ContentService.createTextOutput(JSON.stringify({ok:true,payload:null})).setMimeType(ContentService.MimeType.JSON);const payload=sh.getRange(sh.getLastRow(),3).getValue();return ContentService.createTextOutput(JSON.stringify({ok:true,payload:JSON.parse(payload)})).setMimeType(ContentService.MimeType.JSON);}
